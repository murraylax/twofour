\begin{document}

\frame{\titlepage \setcounter{framenumber}{0}}

\frame
{
  \ft{Motivation and Purpose}
  \uncover<+->{
  \begin{block}{Motivation: Students Transfer}
    \bi
    \item Western Technical College transfer partnerships with UWL, three other UW schools and Viterbo University
    \item UW-Madison has Guaranteed Transfer Programs with all UW system two year colleges and four technical colleges
    \ei
  \end{block}
  }

  \uncover<+->{
    \begin{block}{Purpose}
    Besides academic alignment, measure match of other essential, difficult to measure, abilities:    
    \be
    \item Mindset: Fixed vs. Growth, attitude on ability to learn
    \item Writing ability / confidence
    \item Information-seeking behavior: library research skills
    \ee
  \end{block}
  }
}

\frame
{
  \ft{Dissemination}
  \uncover<+->{
  \begin{block}{Dissemination}
    Submissions of three papers to peer-reviewed journals:
    \be
    \item Mindset
    \item Writing ability / confidence
    \item Information-seeking behavior
    \ee
    Possible target journals:\\ \textit{Higher Education Quarterly}, \textit{International Journal of Higher Education}, \textit{Studies in Higher Education}
  \end{block}
  }

  \uncover<+->{
  \begin{block}{Collaborators}
    Two other collaborators, with equal involvement and workload
    \bi
    \item Sloan Komissarov, Lecturer, Western Technical College
    \item Brenda Murray, Associate Lecturer, UW-La Crosse
    \ei
  \end{block}
  }
}

\frame
{
  \ft{Survey}

  \uncover<+->{
  \begin{block}{Single Survey}
  Single survey to first-year and second-year UWL and WTC students
  \end{block}  
  }

  \uncover<+->{
  \begin{block}{Outcome Variables}
  Survey questions to measure multiple outcome variables related to:
    \vspace*{-1pc}\be
    \item \textit{mindset}
    \item \textit{writing skills}
    \item \textit{information-seeking behavior}
    \ee
  \end{block}
  }

  \uncover<+->{
  \begin{block}{Explanatory and Control Variables}
    \bi
    \item Key variable for comparison: UWL or WTC student
    \item Academic variables: high school performance, college credit experience, credit load, college GPA
    \item Demographics: Race, ethnicity, gender, parents' education
    \item \textit{Interactions effects!} UWL-vs-WTC by demographics
    \ei
  \end{block}
  }
}

\frame
 {
   \ft{Outcome Variables}
   \uncover<+->{
   \begin{block}{Mindset}
   \bi
   \item Mindset predicts academic outcomes (Dweck (2012))
   \item Mindset survey questions: Schmidt et al. (2015) and \texttt{mindsetonline.com} (companion to Dweck (2012))
   \ei  
   \end{block}
   }

   \uncover<+->{
   \begin{block}{Writing Skills}
     \bi
     \item Multiple choice questions from College Board's ACCUPLACER exam
     \item Short-answer questions on purpose, audience, or genre of students' own work
       \ei
   \end{block}
   }

   \uncover<+->{
   \begin{block}{Information-Seeking Behavior}
     \bi
     \item Extends Komissarov and Murray (2016)
     \item Measure student experience using library search tools, academic sources
       \ei
   \end{block}
   }
 }

 \frame
 {
   \ft{Impact}
   \uncover<+->{
   \begin{block}{Inform policy}
     \bi
     \item What should be included in transfer agreements?
     \item What resources should four-year institutions provide for transfer students?
     \item Policies and resources for at-risk populations
     \ei
   \end{block}
   }

   \uncover<+->{
   \begin{block}{Inform Curriculum}
     \bi
     \item Classroom culture to promote growth mindset
     \item Writing and library research assignments
     \ei
   \end{block}
   }
 }
